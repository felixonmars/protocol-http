def call(stream)

Write the body to the given stream.

Implementation

 
def call(stream)
	while chunk = self.read
		stream.write(chunk)
	end
ensure
	stream.close($!)
end

def each

Enumerate all chunks until finished, then invoke #close.

Implementation

 
def each
	while chunk = self.read
		yield chunk
	end
ensure
	self.close($!)
end

def join

Read all remaining chunks into a single binary string using #each.

Implementation

 
def join
	buffer = String.new.force_encoding(Encoding::BINARY)
	
	self.each do |chunk|
		buffer << chunk
		chunk.clear
	end
	
	if buffer.empty?
		return nil
	else
		return buffer
	end
end
